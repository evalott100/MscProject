\section{Glossary of Notation}
\setcounter{page}{1}
	\begin{itemize}
		\item{\makebox[2cm]{$\M_n(F)$\hfill} The set of $n\times n$ matrices with entries in field $F$.}
		\item{\makebox[2cm]{$\I_n$\hfill} The $n\times n$ identity matrix.}
		\item {\makebox[2cm]{$\GL_n(F)$\hfill} The set of invertible $n\times n$ matrices with entries in field $F$.}
		\item {\makebox[2cm]{$\GL(V)$\hfill} The set of linear automorphisms on vector space $V$.}
		\item {\makebox[2cm]{$\Id_X$\hfill} Identity morphism of a space $X$.}
		\item {\makebox[2cm]{$C_n$\hfill} Cyclic group of order $n$.}
		\item {\makebox[2cm]{$D_{2n}$\hfill} Dihedral group of order $2n$.}
		\item {\makebox[2cm]{$S_n$\hfill} Symmetric group of order $n$.}
		\item {\makebox[2cm]{$1_X$\hfill} The identity of a space $X$ with algebraic structure.}
	\end{itemize}
